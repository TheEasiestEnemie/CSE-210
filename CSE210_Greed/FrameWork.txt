Collectable : Object
    float velocity;
    int size;
    var color;
    float x;
    float y;
    var type;

    Collectable()
        velocity = 0
        size = 10
        color = GetRandomColor()
        x = random(0, SCREENSIZEX)
        y = 0
        int coinFlip = Random(0,1);
        if coinFlip == 0
            type = new Rock;
        else if coinFlip == 1
            type = new Gem;

    Collectable(int v, int s, var c, float posx, float posy, var t) //velocity, size, color, x, y, type of collectable. For creating the object yourself
        velocity = v
        size = s
        color = c
        x = posx
        y = posy
        type = t
    
    int tick = 0

    Falling loop
        type.DrawCollectable();
        tick++ //use this to slow down fall speed of collectable.
        y--
        if off screen, delete itself

    DeleteYourself()
        enddrawing();
